import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
  <Card title="Everything Is A Plugin" icon="puzzle">
    Ecewo provides a minimal core by design. Everything else is a plugin, so you only install and use what you need.
  </Card>

	<Card title="Async Support" icon="seti:c">
		Ecewo is built on [libuv](https://github.com/libuv/libuv), a native C library designed for non-blocking asynchronous I/O operations and originally developed for Node.js.
	</Card>

  <Card title="Simple As Node.js" icon="node">
    Inspired by the simplicity of Express.js and plugin architecture of Fastify.
	</Card>

	<Card title="Built-in JSON and CBOR Support" icon="seti:json">
    Ecewo optionally includes [cJSON](https://github.com/DaveGamble/cJSON) for working with JSON and [TinyCBOR](https://github.com/intel/tinycbor) for working with CBOR â€” offered as plugins of course!
	</Card>

  <Card title="Middleware Structure" icon="puzzle">
		Offered Express.js-style route-specific and global middlewares.
	</Card>

  <Card title="Env Configuration" icon="setting">
		Ecewo optionally offers [dotenv-c](https://github.com/Isty001/dotenv-c), a built-in `.env` file parser for C.
	</Card>

  <Card title="Authentication Mechanism" icon="seti:lock">
		You can easily do authentication and authorization using the built-in cookie and session management plugin, or [l8w8jwt](https://github.com/GlitchedPolygons/l8w8jwt) library.
	</Card>

	<Card title="Read The Docs" icon="open-book">
		Start from [Introduction](/docs/introduction) page to read the documentation.
	</Card>

	<Card title="Star on Github" icon="star">
		Star the [Github repo](https://github.com/savashn/ecewo) if you find it useful.
	</Card>

</CardGrid>
