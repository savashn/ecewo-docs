import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid stagger>
  <Card title="Simple As Node.js" icon="node">
    Ecewo is inspired by the simplicity of Express.js. So it's really simple, but in a C kind of way.
	</Card>

	<Card title="Asynchronous Support" icon="seti:c">
		Ecewo is built on [libuv](https://github.com/libuv/libuv), a native C library designed for non-blocking asynchronous I/O operations and originally developed for Node.js.
	</Card>

	<Card title="JSON and CBOR Support" icon="seti:json">
    Both text-based responses such as JSON and binary-based responses such as CBOR are supported.
	</Card>

  <Card title="Middleware Structure" icon="puzzle">
		Provides built-in support for advanced route-level and global middleware, similar to Express.js.
	</Card>

  <Card title="Authentication Mechanism" icon="seti:lock">
		You can easily do authentication and authorization using the built-in cookie and [ecewo-session](https://github.com/savashn/ecewo-session).
	</Card>

  <Card title="Asynchronous Postgres Support" icon="seti:db">
    It's supported to run asynchronous Postgres queries with [pquv](https://github.com/savashn/pquv), a library that integrates [libpq](https://www.postgresql.org/docs/current/libpq.html) with [libuv](https://libuv.org/) to simplify running asynchronous Postgres queries in libuv-based projects.
  </Card>

	<Card title="Read The Docs" icon="open-book">
		Start from [Getting Started](/docs/getting-started) page to read the documentation.
	</Card>

	<Card title="Star on Github" icon="star">
		Star the [Github repo](https://github.com/savashn/ecewo) if you find it useful.
	</Card>

</CardGrid>
